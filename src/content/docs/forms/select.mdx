## API Reference

This component inherits from the [Field Group component](/forms/field-group/).

This component is based on the `<select>` element and can take in any additional HTML attributes through the use of the `inputProps` property. The following are the custom props that can be passed to the component.

<PropTable
  rows={[
    {
      prop: 'name',
      type: 'string',
      description:
        'Unique name of the select. Should follow a kebab-case naming convention (e.g. "first-name")',
      required: true,
    },
    {
      prop: 'label',
      type: 'string',
      description: 'Visual or assistive label for the select',
      required: true,
    },
    {
      prop: 'options',
      type: '[{ value: string, label: string }]',
      description: 'The array of options to display in the dropdown',
      required: true,
    },
    {
      prop: 'labelHidden',
      type: 'boolean',
      description:
        'Whether to visually hide the label or not. When hidden, the label will still be available to screen readers',
      default: 'false',
    },
    {
      prop: 'sublabel',
      type: 'string',
      description: 'Sub-label content for the select',
    },
    {
      prop: 'instructions',
      type: 'string',
      description:
        'Instructions for the select that will be placed below the input',
    },
    {
      prop: 'placeholder',
      type: 'string',
      description: 'Placeholder text for the select',
    },
    {
      prop: 'disabled',
      type: 'boolean',
      description: 'Whether the select is disabled or not',
      default: 'false',
    },
    {
      prop: 'required',
      type: 'boolean | "long"',
      description:
        'Whether the select is required or not. If set to "long", the select will be marked as required and include explicit "required" text.',
      default: 'false',
    },
    {
      prop: 'inputProps',
      type: 'HTMLAttributes<select>',
      description: 'Additional attributes to pass to the select field',
    },
    {
      prop: 'errorMessages',
      type: 'string[]',
      description: 'Error messages to display below the select',
    },

]}
/>
